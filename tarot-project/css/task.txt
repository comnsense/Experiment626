// === ФИЛТРИРАНЕ НА КАРТИ ===
function filterCards() {
    return tarotCards.filter(card => {
        if (currentFilter === 'all') return true;

        // Филтри за бои
        if (['wands', 'cups', 'swords', 'pentacles', 'major'].includes(currentFilter)) {
            return card.suit === currentFilter;
        }

        // Филтри за числа (1-10)
        if (currentFilter.startsWith('number-')) {
            const num = currentFilter.split('-')[1];
            // За малки аркани числото е string (1,2,3...), за големи - римски (I,II,III...)
            return card.number === num || card.number === toRoman(num);
        }

        // Филтри за придворни (на български)
        if (currentFilter.startsWith('court-')) {
            const courtType = currentFilter.split('-')[1];
            
            if (courtType === 'page') {
                // Паж - търсим в cardName за "Паж"
                return card.cardName.includes('Паж') && card.suit !== 'major';
            }
            else if (courtType === 'knight') {
                // Рицар - търсим в cardName за "Рицар"
                return card.cardName.includes('Рицар') && card.suit !== 'major';
            }
            else if (courtType === 'queen') {
                // Царица - търсим в cardName за "Царица"
                return card.cardName.includes('Царица') && card.suit !== 'major';
            }
            else if (courtType === 'king') {
                // Цар - търсим в cardName за "Цар"
                return card.cardName.includes('Цар') && card.suit !== 'major';
            }
        }

        return false;
    });
}

// === ПОМОЩНИ ФУНКЦИИ ЗА КАРТИ ===
function getSuitClass(suit) {
    const classes = {
        'major': 'suit-major',
        'wands': 'suit-wands',
        'cups': 'suit-cups',
        'swords': 'suit-swords',
        'pentacles': 'suit-pentacles'
    };
    return classes[suit] || '';
}

function getSuitName(suit) {
    const names = {
        'major': 'Големи аркани',
        'wands': 'Жезли',
        'cups': 'Купи',
        'swords': 'Мечове',
        'pentacles': 'Пентакли'
    };
    return names[suit] || suit;
}

function getShortKeywords(card) {
    if (card.keywords && card.keywords.length) {
        return card.keywords.slice(0, 3).join(', ');
    }
    return '';
}

/**
 * Клас за тълкуване на таро комбинации
 * Съдържа всички правила за специални комбинации, стихийни баланси и числови модели
 */
class TarotInterpreter {
    constructor() {
        this.rules = this.initAllRules();
    }

    /**
     * Инициализира всички правила за тълкуване
     */
    initAllRules() {
        return {
            // ЧАСТ 1: ОСНОВНИ ПРАВИЛА ЗА СТИХИИ
            elementalRules: [
                {
                    id: "ELEM_FIRE",

                    }

    // ========== ПОМОЩНИ ФУНКЦИИ ==========

    /**
     * Проверява дали конкретна карта присъства
     */
    hasCard(cards, cardId) {
        return cards.some(c => c.id === cardId);
    }

    /**
     * Брой карти от дадена боя
     */
    countSuit(cards, suit) {
        return cards.filter(c => c.suit === suit).length;
    }

    /**
     * Брой карти с дадено число
     */
    countNumber(cards, number) {
        return cards.filter(c => c.number == number).length;
    }

    /**
     * Има ли карта с дадено число
     */
    hasNumber(cards, number) {
        return cards.some(c => c.number == number);
    }

    /**
     * Има ли повтарящо се число
     */
    hasRepeatedNumber(cards, number) {
        return this.countNumber(cards, number) >= 2;
    }

    /**
     * Има ли Асо
     */
    hasAnyAce(cards) {
        return cards.some(c => c.number == "I" || c.number == "1" || c.number == "0");
    }

    /**
     * Има ли 10-ца
     */
    hasAnyTen(cards) {
        return cards.some(c => c.number == "X" || c.number == "10");
    }

    /**
     * Взема числото от карта (като инт)
     */
    getCardNumber(card) {
        const numMap = {
            "0": 0, "I": 1, "II": 2, "III": 3, "IV": 4, "V": 5, "VI": 6, "VII": 7, "VIII": 8, "IX": 9, "X": 10,
            "XI": 11, "XII": 12, "XIII": 13, "XIV": 14, "XV": 15, "XVI": 16, "XVII": 17, "XVIII": 18, "XIX": 19, "XX": 20, "XXI": 21
        };
        return numMap[card.number] || parseInt(card.number) || 0;
    }

    // ========== ОСНОВНА ТЪЛКУВАТЕЛНА ФУНКЦИЯ ==========

    /**
     * Тълкува списък от карти според всички правила
     * @param {Array} cards - Масив от карти (обекти)
     * @param {string} questionType - Тип въпрос: 'love', 'career', 'home', 'finance', 'spiritual', 'general'
     * @returns {Object} - Резултат от тълкуването
     */
    interpret(cards, questionType = 'general') {
        const result = {
            elementalTheme: null,
            numberMessages: [],
            specialCombinations: [],
            questionAnswer: null,
            summary: "",
            advice: ""
        };

        // 1. Провери стихийни правила
        for (const rule of this.rules.elementalRules) {
            if (rule.condition(cards)) {
                result.elementalTheme = rule.result;
                break;
            }
        }

        // 2. Провери числови правила
        for (const rule of this.rules.numberRules) {
            if (rule.condition(cards)) {
                result.numberMessages.push(rule.result);
            }
        }

        // 3. Провери специални комбинации според типа въпрос
        let rulesToCheck = [];

        switch(questionType) {
            case 'love':
                rulesToCheck = this.rules.loveRules;
                break;
            case 'career':
                rulesToCheck = this.rules.careerRules;
                break;
            case 'home':
                rulesToCheck = this.rules.homeRules;
                break;
            case 'finance':
                rulesToCheck = this.rules.financeRules;
                break;
            case 'spiritual':
                rulesToCheck = this.rules.spiritualRules;
                break;
            default:
                // Ако е general, проверяваме всички
                rulesToCheck = [
                    ...this.rules.loveRules,
                    ...this.rules.careerRules,
                    ...this.rules.homeRules,
                    ...this.rules.financeRules,
                    ...this.rules.spiritualRules,
                    ...this.rules.warningRules,
                    ...this.rules.blessingRules,
                    ...this.rules.elementalCombos,
                    ...this.rules.numberCombos,
                    ...this.rules.majorArcanaCombos
                ];
        }

        for (const rule of rulesToCheck) {
            if (rule.condition && rule.condition(cards)) {
                result.specialCombinations.push(rule.result || rule);
            }
        }

        // 4. Провери предупредителни правила (винаги)
        for (const rule of this.rules.warningRules) {
            if (rule.condition(cards)) {
                result.specialCombinations.push({ ...rule.result, warning: true });
            }
        }

        // 5. Провери щастливи правила (винаги)
        for (const rule of this.rules.blessingRules) {
            if (rule.condition(cards)) {
                result.specialCombinations.push({ ...rule.result, blessing: true });
            }
        }

        // 6. Ако има конкретен въпрос, провери questionRules
        if (questionType !== 'general' && this.rules.questionRules[questionType]) {
            for (const rule of this.rules.questionRules[questionType]) {
                if (rule.condition(cards)) {
                    result.questionAnswer = rule.answer;
                    break;
                }
            }
        }

        // 7. Генерирай обобщение
        result.summary = this.generateSummary(result, cards);
        
        return result;
    }

    /**
     * Генерира обобщение от всички намерени комбинации
     */
    generateSummary(result, cards) {
        let summary = "";
        
        if (result.elementalTheme) {
            summary += `🌍 Доминираща стихия: ${result.elementalTheme.theme}\n`;
            summary += `   💡 ${result.elementalTheme.advice}\n`;
        }
        
        if (result.numberMessages.length > 0) {
            summary += `🔢 Числови послания: ${result.numberMessages.join('; ')}\n`;
        }
        
        if (result.specialCombinations.length > 0) {
            summary += `🔮 Специални комбинации:\n`;
            result.specialCombinations.forEach((combo, index) => {
                if (typeof combo === 'string') {
                    summary += `  ${index+1}. ${combo}\n`;
                } else {
                    summary += `  ${index+1}. ${combo.type || 'Комбинация'}: ${combo.description || combo}\n`;
                    if (combo.advice) {
                        summary += `     💡 Съвет: ${combo.advice}\n`;
                    }
                }
            });
        }
        
        if (result.questionAnswer) {
            summary += `❓ Отговор на въпроса: ${result.questionAnswer}\n`;
        }
        
        if (cards.length >= 3) {
            // Времева линия за първите 3 карти
            summary += `⏳ Времева линия:\n`;
            summary += `  Минало (основа): ${cards[0]?.cardName || cards[0]?.name || '—'}\n`;
            summary += `  Настояще (предизвикателство): ${cards[1]?.cardName || cards[1]?.name || '—'}\n`;
            summary += `  Бъдеще (резултат): ${cards[2]?.cardName || cards[2]?.name || '—'}\n`;
        }
        
        return summary;
    }

    /**
     * Бърз отговор ДА/НЕ на базата на картите
     */
    quickYesNo(cards) {
        const positiveCards = ['sun', 'world', 'ace_cups', 'ace_pentacles', 'six_wands', 'ten_cups', 'ace_wands', 'six_cups'];
        const negativeCards = ['tower', 'death', 'three_swords', 'five_pentacles', 'ten_swords', 'devil', 'five_cups', 'nine_swords'];
        
        let score = 0;
        
        cards.forEach(card => {
            if (positiveCards.includes(card.id)) score += 2;
            if (negativeCards.includes(card.id)) score -= 2;
            
            // Ако има обърнати карти (опростено - тук приемаме, че всички са изправени)
            // В реално приложение трябва да имаме информация за позицията
        });
        
        // Провери за силни комбинации
        if (this.hasCard(cards, 'sun') && this.hasCard(cards, 'ten_cups')) score += 3;
        if (this.hasCard(cards, 'tower') && this.hasCard(cards, 'five_pentacles')) score -= 3;
        
        if (score >= 3) return "ДА ✅";
        if (score <= -3) return "НЕ ❌";
        if (score > 0) return "По-скоро ДА";
        if (score < 0) return "По-скоро НЕ";
        return "НЕ Е ЯСНО - зависи от теб";
    }

    /**
     * Търси карта по име или част от име
     */
    searchCards(searchTerm) {
        const term = searchTerm.toLowerCase();
        return tarotCards.filter(card => 
            card.cardName.toLowerCase().includes(term) || 
            card.name.toLowerCase().includes(term) ||
            (card.keywords && card.keywords.some(k => k.toLowerCase().includes(term)))
        );
    }
}
